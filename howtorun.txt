1. instalacja ubuntu/debian
2. instalacja ftp "sudo apt install vsftpd"
3. instalacja apache "sudo apt install apache2"
4. instalacja mariadb "sudo apt install mariadb-server mariadb-client -y"
5. instalacja php "sudo apt install php php-common php-mysql php-gd php-cli -y", a następnie
	stworzenie pliku info.php "echo "<?php phpinfo(); ?>" | sudo tee /var/www/html/info.php",
	aby można było podglądać wersję php: http://SERVER_IP/info.php
6. instalacja phpmyadmin "sudo apt install phpmyadmin"
7. Dodanie w pliku: /etc/apache2/apache2.conf linijki: "Include /etc/phpmyadmin/apache.conf"
8. restart apache "sudo systemctl restart apache2"
9. logowanie do mariadb "sudo mysql -u root -p", a następnie utworzenie admina:
	MariaDB [(none)]> CREATE USER 'admin’@’localhost' IDENTIFIED BY 'hostnextra=@3454';
	MariaDB [(none)]> GRANT ALL PRIVILEGES ON *.* TO 'admin’@’localhost' WITH GRANT OPTION;
	MariaDB [(none)]> FLUSH PRIVILEGES;
	
	albo za to środkowe:
	GRANT ALL PRIVILEGES ON *.* TO admin@localhost IDENTIFIED BY ' your current password ' WITH GRANT

21. Przygotowanie venv

Jeśli w systemie nie ma python3.8 to trzeba najpierw zainstalować:

sudo apt update
sudo apt install build-essential zlib1g-dev libncurses5-dev libgdbm-dev libnss3-dev libssl-dev libsqlite3-dev libreadline-dev libffi-dev curl libbz2-dev
curl -O https://www.python.org/ftp/python/3.8.2/Python-3.8.2.tar.xz
tar -xf Python-3.8.2.tar.xz
cd Python-3.8.2
./configure --enable-optimizations --enable-loadable-sqlite-extensions
make -j 2
Modify the -j to correspond to the number of cores in your processor. You can find the number by typing nproc.

sudo make altinstall
Do not use the standard make install as it will overwrite the default system python3 binary.

27. Konfiguracja apache (sprawdzanie konfiguracji - sudo apachectl configtest)

sudo nano /etc/apache2/apache2.conf

Zmienić domyślne
<Directory /var/www>
	Options Indexes FollowSymLinks Includes ExecCGI
	AllowOverride None
	Require all granted
</Directory>
na 
<Directory /intranet/data>
	Options Indexes FollowSymLinks Includes ExecCGI
	AllowOverride None
	Require all granted
</Directory>


sudo nano /etc/apache2/sites-enabled/000-default.conf

<VirtualHost *:80>
	RewriteEngine On
	RewriteCond %{HTTPS} !=on
	RewriteRule ^/?(.*) https://%{SERVER_NAME}/$1 [R=301,L]
</VirtualHost>

<VirtualHost *:443>
	DocumentRoot /intranet/data/
	ServerName intranet.domain.pl
	
	Alias /static/ /intranet/data/static/
	Alias /media/ /intranet/data/media/
	Alias /shared/ /intranet/data/network_shared_files/
	
	<Directory /intranet/static>
		Options -Indexes
		AllowOverride None
		Order allow,deny
		Allow from all
	</Directory>
	<Directory /intranet/media>
		Options -Indexes
		AllowOverride None
		Order allow,deny
		Allow from all
	</Directory>
	<Directory /intranet/data/network_shared_files>
		Options -Indexes
		AllowOverride None
		Order allow,deny
		Allow from all
	</Directory>

	RewriteEngine On
	RewriteCond %{HTTP:Connection} Upgrade		[NC]
	RewriteCond %{HTTP:Upgrade} websocket		[NC]
	RewriteRule /(.*) ws://127.0.0.1:8001/$1	[P,L]

	RequestHeader set X-Forwarded-Proto 'https' env=HTTPS

	ProxyPass /static !
	ProxyPass /media !
	ProxyPass /shared !
	ProxyPass /phpmyadmin !
	ProxyRequests Off
	ProxyPreserveHost On
	ProxyPass / http://0.0.0.0:8001/
	ProxyPassReverse / http://0.0.0.0:8001/
	
	SSLEngine on
	SSLCertificateFile /etc/apache2/certificate/cert.txt
	SSLCertificateKeyFile /etc/apache2/certificate/private.txt
	
	ErrorLog ${APACHE_LOG_DIR}/error.log
	CustomLog ${APACHE_LOG_DIR}/access.log combined
</VirtualHost>

sudo systemctl restart apache2
